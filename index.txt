<script>
function generateVCard() {
  const name = document.getElementById("name").value;
  const title = document.getElementById("title").value;
  const phone = document.getElementById("phone").value;
  const email = document.getElementById("email").value;
  const company = document.getElementById("company").value;
  const website = document.getElementById("website").value;

  const vcard = `BEGIN:VCARD\nVERSION:3.0\nN:${name}\nTITLE:${title}\nTEL:${phone}\nEMAIL:${email}\nORG:${company}\nURL:${website}\nEND:VCARD`;

  const encoded = encodeURIComponent(vcard);
  const qrURL = `https://api.qrserver.com/v1/create-qr-code/?size=250x250&data=${encoded}`;

  document.getElementById("output").innerHTML = `
    <div class='vcard-preview'>
      <h3>${name}</h3>
      <p>${title}</p>
      <p>${company}</p>
      <p>${phone}</p>
      <p>${email}</p>
      <p>${website}</p>

      <div class='qr'>
        <img src='${qrURL}' width='200' />
      </div>

      <br>
      <a href="data:text/vcard;charset=utf-8,${encoded}" download="vcard.vcf" style="text-decoration:none">
        <button>Download V-Card File</button>
      </a>
    </div>
  `;

  document.getElementById("output").style.display = "block";
}
</script>
